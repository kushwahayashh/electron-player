Playback Feedback Component

This is a self-contained component that displays visual feedback when a video is played, paused, muted, or unmuted. The feedback icon appears in the center of the player, scales up, and then fades away.

Video Player Area
Integration Instructions
/* Add this CSS to your main stylesheet */
.playback-feedback-overlay {
  position: absolute;
  top: 0;
  left: 0;
  width: 100%;
  height: 100%;
  display: flex;
  align-items: center;
  justify-content: center;
  pointer-events: none;
  z-index: 10;
}

.feedback-icon {
  font-size: 80px;
  color: rgba(255, 255, 255, 0.9);
  opacity: 0;
  transform: scale(0.8);
  transition: none;
}

.feedback-icon.animate {
  animation: feedbackAnimation 0.6s ease-out forwards;
}

@keyframes feedbackAnimation {
  0% {
    opacity: 0;
    transform: scale(0.8);
  }
  30% {
    opacity: 1;
    transform: scale(1.2);
  }
  100% {
    opacity: 0;
    transform: scale(1.4);
  }
}
/* Add this JavaScript to your player controller */
showPlaybackFeedback(isPlaying) {
  // Create or get the feedback overlay
  let overlay = this.elements.player.querySelector('.playback-feedback-overlay');
  
  // If overlay doesn't exist, create it
  if (!overlay) {
    overlay = document.createElement('div');
    overlay.className = 'playback-feedback-overlay';
    this.elements.player.appendChild(overlay);
  }
  
  // Create or get the feedback icon
  let icon = overlay.querySelector('.feedback-icon');
  if (!icon) {
    icon = document.createElement('i');
    icon.className = 'feedback-icon';
    overlay.appendChild(icon);
  }
  
  // Set the icon based on play state
  icon.className = 'feedback-icon ' + 
    (isPlaying ? 'ti ti-player-play-filled' : 'ti ti-player-pause-filled');
  
  // Reset and trigger animation
  icon.classList.remove('animate');
  void icon.offsetWidth; // Trigger reflow
  icon.classList.add('animate');
}

showVolumeFeedback(isMuted) {
  // Create or get the feedback overlay
  let overlay = this.elements.player.querySelector('.playback-feedback-overlay');
  
  // If overlay doesn't exist, create it
  if (!overlay) {
    overlay = document.createElement('div');
    overlay.className = 'playback-feedback-overlay';
    this.elements.player.appendChild(overlay);
  }
  
  // Create or get the feedback icon
  let icon = overlay.querySelector('.feedback-icon');
  if (!icon) {
    icon = document.createElement('i');
    icon.className = 'feedback-icon';
    overlay.appendChild(icon);
  }
  
  // Set the icon based on mute state
  icon.className = 'feedback-icon ' + 
    (isMuted ? 'ti ti-volume-off' : 'ti ti-volume');
  
  // Reset and trigger animation
  icon.classList.remove('animate');
  void icon.offsetWidth; // Trigger reflow
  icon.classList.add('animate');
}

How to integrate:

  1. Add the CSS from the first code block to your main stylesheet
  2. Add both JavaScript methods to your VideoPlayerController class
  3. Call showPlaybackFeedback(true) when playing and showPlaybackFeedback(false) when pausing
  4. Call showVolumeFeedback(true) when muting and showVolumeFeedback(false) when unmuting
